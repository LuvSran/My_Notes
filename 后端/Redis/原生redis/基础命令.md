```c++
select k //(切换到第k个数据库,0~15)

keys * //查看所有[key] 

keys prefix* //查看prefix开头的[key]

move a_key k //将a_key[key]移动到第k个数据库

persist a_key //取消a_key[key]的过期时间

type a_key //查看a_key[key]的类型

client list //查看链接到服务器的客户端列表

dbsize //查看当前数据库[key]数量

flushdb //删除当前数据库所有[key]

flushall //删除所有数据库所有[key]
```
