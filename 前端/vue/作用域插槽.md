子
```js
//<!-- UserList.vue -->
<template>

  <div class="list-container">
    <div v-for="user in users" :key="user.id">
      <slot name="item" :user="user" :index="user.id"></slot>
    </div>
  </div>

</template>
<script setup>
import { ref } from 'vue'
// 子组件内部的数据
const users = ref([
  { id: 1, name: '张三', age: 24 },
  { id: 2, name: '李四', age: 25 }
])
</script>
```
父
```js
//<!-- Parent.vue --> 
<template>
 <UserList>
   // slotProps 包含了子组件传递的所有属性（user, index）
    <template #item="{ user, index }">
      <div class="custom-card">
        <h3>{{ index }} - {{ user.name }}</h3>
        <p v-if="user.age > 24">这是老员工</p>
      </div>
    </template>
  </UserList>
 </template>
```
