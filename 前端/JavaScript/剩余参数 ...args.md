**`...args`（剩余参数）是真正的数组。**
```js
// 1. 它是真数组，可以直接用 reduce 求和
function sum(...nums) {
  return nums.reduce((a, b) => a + b, 0);
}
console.log(sum(1, 2, 3)); // 输出: 6

// 2. 它可以和普通参数混用
function log(first, ...rest) {
  console.log('第一个参数:', first);       // "A"
  console.log('剩余参数:', rest);         // ["B", "C"] (是数组)
  console.log('能否使用 map:', rest.map(x => x.toLowerCase())); // 可以!
}
log('A', 'B', 'C');


// 3. 它不能在 arguments 中使用的箭头函数里工作
const arrowFunc = (...args) => {
  console.log(args.length); // 正常输出参数个数
};
arrowFunc(1, 2, 3);
```
