class eventEmitter{
  constructor(){
    this.events={};
  }

  on(e,fn){
    if(!this.events[e]) this.events[e]=[];
    this.events[e].push(fn);
  }

  off(e,fn){
    const handler=this.events[e];
    if(!handler) return;
    const idx=handler.indexOf(fn);
    if(idx === -1) return;
    handler.splice(idx,1);
  }

  emit(e,...args){
    const handler=this.events[e];
    if(!handler) return;
    handler.slice().forEach((fn)=>{
      fn(...args);
    });
  }
}
