**浅拷贝**
```js
function shallow(obj){
  if(!obj || typeof obj !== 'object') return obj;
  const newObj=Array.isArray(obj)?[]:{};
  for(let key in obj){
    if(obj.hasOwnProperty(key)) newObj[key]=obj[key];
  } //hasOwnProperty判断一个属性是否为自有属性,而非原型链继承属性。
  return newObj;
}
```
**深拷贝**
```js
function deepCopy(obj,mp=new WeakMap()){
  if(!obj || typeof obj !=='object') return obj;
  if(obj instanceof Date) return new Date(obj);
  if(obj instanceof RegExp) return new RegExp(obj);
  if(mp.has(obj)) return mp.get(obj);
  const newObj=Array.isArray(obj)?[]:{};
  mp.set(obj,newObj);
  for(const key in obj){
    if(obj.hasOwnProperty(key)){
      newObj[key]=deepCopy(obj[key],mp);
    }
  }
  return newObj;
}
```
*进阶:* 不能拷贝 symbol 类型的字段，因为for..in 无法遍历到 symbol 的键。改进需要用`Reflect.ownKeys(obj).forEach(key => { newObj[key] = deepCopy(obj[key], mp); });`
