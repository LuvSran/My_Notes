**时间戳版**
一次触发会立即执行，但最后一次触发如果不在时间周期内，则不会执行。
```js
function throttle(fn,delay){
  let lastTime=0;
  const _throttle=function(...args){
    const now=Date.now(); //若要精确到微秒则用performance.now();
    if(now-lastTime>=delay){
      fn.apply(this,args);
      lastTime=now;
    }
  };
  return _throttle;
}
```
**定时器版**
第一次触发不会立即执行（会延迟 `delay` ），但停止触发后，最后一次操作保证会执行一次。
```js
function throttle(fn,delay){
  let timer=null;
  const _throttle=function(...args){
    if(!timer){
      timer=setTimeout(()=>{
        fn.apply(this,args);
        timer=null;
      },delay);
    }
  }
  return _throttle;
}
```
