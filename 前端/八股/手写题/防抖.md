```js
function debounce(fn, delay){
  let timer = null
  const _debounce = function(...args){
    if(timer) clearTimeout(timer)
    timer = setTimeout(()=>{
      fn.apply(this, args)
    }, delay)
  }
  return _debounce
}
```

```js
function debounce(fn, delay, immediate = false) {
  let timer = null; 
  const _debounce=function(...args) {
    if (timer) clearTimeout(timer);
    if (immediate) {
      const call = !timer;
      timer = setTimeout(() => {
        timer = null;
      }, delay);
      if (call) {
        fn.apply(this, args);
      }
    }
    else {
      timer = setTimeout(() => {
        fn.apply(this, args);
      }, delay);
    }
  };
  return _debounce;
}
```
