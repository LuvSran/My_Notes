并发池
```go
package main  
import (  
    "fmt"  
    "sync"
)    
var wg sync.WaitGroup  
func main() {  
    jobChan := make(chan int64, 20)  
    resChan := make(chan int, 1000)  
    get := func(u int64) int {  
       sum := 0  
       for ; u != 0; u /= 10 {  
          sum += int(u % 10)  
       }  
       return sum  
    }  
    for i := 1; i <= 24; i++ {    //worker Pool建立  
       wg.Add(1)  
       go func() {  
          defer wg.Done()  
          for val := range jobChan {  
             resChan <- get(val)  
          }  
       }()  
    }  
    wg.Add(1)   
    go func() {             //启动Producer  
       defer wg.Done()  
       for i := 1; i <= 1000; i++ {  
          jobChan <- int64(i)  
       }  
       close(jobChan)  
    }()  
    wg.Wait()  
    close(resChan)  
    for val := range resChan {  
       fmt.Println(val)  
    }  
}
```
